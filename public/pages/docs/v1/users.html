All users

Get all users from the database.

method: GET

endpoint: /api/v1/users

example:

                
                    function sendRequest() {
                        var xhr = new XMLHttpRequest();
                        xhr.open("GET", "http://localhost:55600/api/v1/users");
                        xhr.setRequestHeader("Content-Type", "application/json");
                        xhr.setRequestHeader("Authorization", "Bearer APIKEY_HERE");
                        xhr.onreadystatechange = function () {
                            if (xhr.readyState === 4) {
                            if (xhr.status === 200) {
                                var response = JSON.parse(xhr.responseText);
                                document.getElementById("response").innerHTML = JSON.stringify(response, null, 2);
                            } else {
                                document.getElementById("response").innerHTML = "Error: " + xhr.status;
                            }
                            }
                        };
                        xhr.send();
                    }

Response:

                
                    [{"id":1,"email":"test@example.com","email_verified":0,"first_name":"John","last_name":"Doe","admin",0,"times_rented":8,"currently_renting":2}]
